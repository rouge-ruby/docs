def stream(tokens, &b)
  lineno = 0
  token_lines(tokens) do |line_tokens|
    yield %(<#{@tag_name} class="#{sprintf @class_format, lineno += 1}">)
    @formatter.stream(line_tokens) {|formatted| yield formatted }
    yield %(\n</#{@tag_name}>)
  end
end
